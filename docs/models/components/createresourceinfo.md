# CreateResourceInfo Object
### Description
The `CreateResourceInfo` object is used to store creation info for a resource.

### Usage Example
Use in the `POST Resources` endpoint.

## Example Usage

```typescript
import { CreateResourceInfo } from "opal-mcp/models/components";

let value: CreateResourceInfo = {
  name: "Okta Engineering Super Admin",
  description: "Okta super admin role.",
  resourceType: "OKTA_ROLE",
  appId: "f454d283-ca87-4a8a-bdbb-df212eca5353",
  remoteResourceId:
    "API_ACCESS_MANAGEMENT_ADMIN-51d203da-313a-4fd9-8fcf-420ce6312345",
  metadata: "{\n"
    + "  \"okta_directory_role\":\n"
    + "    {\n"
    + "      \"role_id\": \"SUPER_ADMIN-b52aa037-4a35-4ac3-9350-f6260fd12345\",\n"
    + "      \"role_type\": \"SUPER_ADMIN\",\n"
    + "    },\n"
    + "}",
};
```
